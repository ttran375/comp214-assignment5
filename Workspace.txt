BEGIN
    FOR r IN (
        SELECT
            table_name
        FROM
            user_tables
    ) LOOP
        BEGIN
            EXECUTE IMMEDIATE 'DROP TABLE "'
                              || r.table_name
                              || '" CASCADE CONSTRAINTS PURGE';
        EXCEPTION
            WHEN OTHERS THEN
                DBMS_OUTPUT.PUT_LINE('Failed to drop table "'
                                     || r.table_name
                                     || '": '
                                     || SQLERRM);
        END;
    END LOOP;

    FOR r IN (
        SELECT
            sequence_name
        FROM
            user_sequences
    ) LOOP
        BEGIN
            EXECUTE IMMEDIATE 'DROP SEQUENCE "'
                              || r.sequence_name
                              || '"';
        EXCEPTION
            WHEN OTHERS THEN
                DBMS_OUTPUT.PUT_LINE('Failed to drop sequence "'
                                     || r.sequence_name
                                     || '": '
                                     || SQLERRM);
        END;
    END LOOP;
END;

CREATE TABLE gpnr_shipment (
    cart_id NUMBER(10) PRIMARY KEY,
    shipping_date DATE,
    total NUMBER(12, 2),
    shipper_username VARCHAR2(20),
    promotion_id NUMBER(10),
    district_id NUMBER(10),
    payment_method_id NUMBER(10),
    FOREIGN KEY (cart_id) REFERENCES gpnr_cart(cart_id),
    FOREIGN KEY (shipper_username) REFERENCES gpnr_shipper(shipper_username),
    FOREIGN KEY (promotion_id) REFERENCES gpnr_promotion(promotion_id),
    FOREIGN KEY (district_id) REFERENCES gpnr_district(district_id),
    FOREIGN KEY (payment_method_id) REFERENCES gpnr_payment_method(payment_method_id)
);

DECLARE
    v_cart_id gpnr_cart.cart_id%TYPE;
    v_shipping_date gpnr_shipment.shipping_date%TYPE;
    v_total gpnr_shipment.total%TYPE;
    v_shipper_username gpnr_shipment.shipper_username%TYPE;
    v_promotion_id gpnr_shipment.promotion_id%TYPE;
    v_district_id gpnr_shipment.district_id%TYPE;
    v_payment_method_id gpnr_shipment.payment_method_id%TYPE;
    v_district_tax gpnr_district.tax%TYPE;
    v_shipping_cost gpnr_district.shipping_cost%TYPE;
    v_payment_discount gpnr_payment_method.payment_method_discount%TYPE;
    v_cart_total NUMBER(12, 2);
    v_promotion_discount NUMBER(12, 2);
BEGIN
    FOR cart_rec IN (SELECT cart_id FROM gpnr_cart WHERE cart_status = 'active')
    LOOP
        -- Calculate total for the cart
        v_cart_total := 0;
        FOR cart_prod_rec IN (SELECT product_id, cart_product_quantity FROM gpnr_cart_product WHERE cart_id = cart_rec.cart_id)
        LOOP
            SELECT price * cart_prod_rec.cart_product_quantity
            INTO v_cart_total
            FROM gpnr_product
            WHERE product_id = cart_prod_rec.product_id;
        END LOOP;
        
        -- Choose random shipper username
        SELECT shipper_username
        INTO v_shipper_username
        FROM (SELECT shipper_username FROM gpnr_shipper ORDER BY DBMS_RANDOM.VALUE)
        WHERE ROWNUM = 1;
        
        -- Fetch random promotion, district, and payment method
        SELECT promotion_id
        INTO v_promotion_id
        FROM (SELECT promotion_id FROM gpnr_promotion ORDER BY DBMS_RANDOM.VALUE)
        WHERE ROWNUM = 1;
        
        SELECT district_id, tax, shipping_cost
        INTO v_district_id, v_district_tax, v_shipping_cost
        FROM (SELECT district_id, tax, shipping_cost FROM gpnr_district ORDER BY DBMS_RANDOM.VALUE)
        WHERE ROWNUM = 1;
        
        SELECT payment_method_id, payment_method_discount
        INTO v_payment_method_id, v_payment_discount
        FROM (SELECT payment_method_id, payment_method_discount FROM gpnr_payment_method ORDER BY DBMS_RANDOM.VALUE)
        WHERE ROWNUM = 1;
        
        -- Calculate promotion discount
        SELECT promotion_discount
        INTO v_promotion_discount
        FROM gpnr_promotion
        WHERE promotion_id = v_promotion_id;
        
        -- Calculate total including taxes, shipping, and payment discount
        v_total := v_cart_total * (1 - v_promotion_discount) * (1 + v_district_tax) + v_shipping_cost;
        v_total := v_total * (1 - v_payment_discount);
        
        -- Insert into gpnr_shipment table
        INSERT INTO gpnr_shipment (cart_id, shipping_date, total, shipper_username, promotion_id, district_id, payment_method_id)
        VALUES (cart_rec.cart_id, SYSDATE, v_total, v_shipper_username, v_promotion_id, v_district_id, v_payment_method_id);
        
        -- Commit after each cart
        COMMIT;
    END LOOP;
END;
/

select * from gpnr_shipment;
